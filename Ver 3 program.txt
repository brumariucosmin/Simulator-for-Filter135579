    methods (Access = public)
        
        function  functie_butterworth_lowpass(app) 
            
            Ordin = app.OrdinEditField.Value;
            Fc = app.FcEditField.Value;
            Fs = app.FsEditField.Value;
                  
            [zb,pb,kb] = butter(Ordin,2*pi*Fc,'s');
            [bb,ab] = zp2tf(zb,pb,kb);
            [hb,wb] = freqs(bb,ab,Fs);
            plot(app.UIAxes,wb/(2e9*pi),mag2db(abs(hb)));
            xlabel(app.UIAxes,'Frequency ')
            ylabel(app.UIAxes,'Attenuation ')
            title(app.UIAxes,'Frequency vs Attenuation')
           
        end
    end

        function simulare_circuit_1(app)
       
        R1 = app.R1EditField.Value;
        R2 = app.R2EditField.Value;
        R3 = app.R3EditField.Value;
        V1 = app.V1EditField.Value;
        double Vc;
        Vo = ((R2 + R3)/R2)*V1;
        Vn = (R3/(R2 + R3))*Vo;        
        Vp = (R3/(R2 + R3))*Vo; 
         app.TextArea.Value{1} = sprintf('Vo e egal cu ')
         app.TextArea.Value{2} = sprintf('%.3f',Vo);  % converts from numeric to char              
         app.TextArea.Value{3} = sprintf('Vn e egal cu')
         app.TextArea.Value{4} = sprintf('%.3f',Vn);  % converts from numeric to char 
         app.TextArea.Value{5} = sprintf('Vp e egal cu')
         app.TextArea.Value{6} = sprintf('%.3f',Vp);  % converts from numeric to char 
         
         x = linspace(0.01, 2*pi, 1500);
         period = pi/4;
         y = Vo* sin(2 * pi * x / period);
         
         
         x1 = linspace(0.01, 2*pi, 1500);
         period = pi/4;
         y1 = Vn* sin(2 * pi * x1 / period);
         
         
         x2 = linspace(0.01, 2*pi, 1500);
         period = pi/4;
         y2 = Vp* sin(2 * pi * x2 / period);
         plot(app.UIAxes ,x2, y2);
         hold on;
         plot(app.UIAxes ,x1, y1);
         plot(app.UIAxes ,x, y);
         
         xlabel(app.UIAxes,'time (s)')
         ylabel(app.UIAxes,'voltage (V)')
         title(app.UIAxes,'Voltage vs time')
        
        end
    end
